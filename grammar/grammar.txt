PROGRAM -> TOP_LEVEL*

TOP_LEVEL -> STATE | DEFINE

STATE -> ["initial"] "state" IDENTIFIER "{" CASE* "}"
DEFINE -> "define" IDENTIFIER (CHARACTER | STRING)

CASE -> (CHARACTER | IDENTIFIER) "," (CHARACTER | IDENTIFIER) "," DIRECTION "->" IDENTIFIER | "self"

IDENTIFIER -> (_ | ALPHABETICAL)(_ | ALPHABETICAL | NUMERICAL)*
DIRECTION -> "L" | "R"

ALPHABETICAL -> a-z | A-Z
NUMERICAL -> 0-9
